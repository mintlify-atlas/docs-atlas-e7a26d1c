## Overview

The Raids System provides a comprehensive dungeon/arena framework with **wave-based combat**, **party management**, **NPC spawning**, and **reward distribution**. It integrates deeply with CustomNPC to create challenging PvE encounters with configurable difficulty and automated session management.

**Location:** `raids/managers/NPCSpawnManager.java`

```java
public class NPCSpawnManager {
    
    // Track NPCs by wave
    private static Map<String, Set<Integer>> waveNpcs = new ConcurrentHashMap<>();
    // Map NPC entity ID → wave ID
    private static Map<Integer, String> npcToWave = new ConcurrentHashMap<>();
    // Track alive status
    private static Map<Integer, Boolean> npcAliveStatus = new ConcurrentHashMap<>();
    
    /**
     * Spawn all NPCs for a wave
     */
    public static boolean spawnWaveNpcs(Wave wave, String waveId) {
        Set<Integer> spawnedNpcs = new HashSet<>();
        
        for (SpawnPoint spawnPoint : wave.getSpawnPoints()) {
            Location loc = spawnPoint.getLocation();
            String npcName = spawnPoint.getNpcName();
            int count = spawnPoint.getCount();
            
            for (int i = 0; i < count; i++) {
                // Spawn NPC using CustomNPC API
                ICustomNpc npc = NpcAPI.Instance().spawnNPC(
                    npcName, 
                    loc.getWorld().getName(),
                    loc.getX(), loc.getY(), loc.getZ()
                );
                
                if (npc != null) {
                    int entityId = npc.getEntityId();
                    spawnedNpcs.add(entityId);
                    npcToWave.put(entityId, waveId);
                    npcAliveStatus.put(entityId, true);
                }
            }
        }
        
        waveNpcs.put(waveId, spawnedNpcs);
        return !spawnedNpcs.isEmpty();
    }
    
    /**
     * Mark NPC as dead
     */
    public static boolean markNpcDead(int entityId, String waveId) {
        if (!npcAliveStatus.containsKey(entityId)) return false;
        npcAliveStatus.put(entityId, false);
        return true;
    }
    
    /**
     * Get alive NPCs count for wave
     */
    public static int getAliveNpcsCount(String waveId) {
        Set<Integer> npcs = waveNpcs.get(waveId);
        if (npcs == null) return 0;
        
        return (int) npcs.stream()
            .filter(id -> npcAliveStatus.getOrDefault(id, false))
            .count();
    }
}
```

public void onNpcDie(INpcEvent.DiedEvent event) {
    // 1. Get NPC entity ID
    int entityId = event.getNpc().getEntityId();
    
    // 2. Find associated wave
    String waveId = NPCSpawnManager.getWaveIdForNpc(entityId);
    if (waveId == null) return; // Not a raid NPC
    
    // 3. Get raid session
    RaidSession session = getSessionByWaveId(waveId);
    if (session == null) return;
    
    // 4. Mark NPC as dead
    boolean wasTracked = NPCSpawnManager.markNpcDead(entityId, waveId);
    if (!wasTracked) return;
    
    // 5. Play death effects
    playDeathEffects(session, killer);
    
    // 6. Check if wave is complete
    int remaining = NPCSpawnManager.getAliveNpcsCount(waveId);
    
    if (remaining == 0) {
        completeWave(session, waveId);
    } else {
        updateWaveProgress(session, remaining);
    }
}

private void completeWave(RaidSession session, String waveId) {
    int waveNumber = session.getCurrentWaveIndex() + 1;
    int totalWaves = session.getRaid().getTotalWaves();
    boolean isLastWave = !session.hasNextWave();
    
    // 1. Notify players
    List<Player> players = getActivePlayers(session);
    for (Player player : players) {
        player.sendMessage("");
        player.sendMessage("§8▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬");
        
        if (isLastWave) {
            player.sendMessage("§6§l  🎉 ¡ÚLTIMA OLEADA COMPLETADA! 🎉");
        } else {
            player.sendMessage(String.format(
                "§a§l  ✓ OLEADA %d/%d COMPLETADA ✓", waveNumber, totalWaves
            ));
        }
        
        player.sendMessage("§8▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬▬");
        player.sendMessage("");
        
        RaidSoundManager.playWaveCompleteSound(player);
    }
    
    // 2. Execute rewards
    executeWaveRewardsOnce(session);
    
    // 3. Clear tracking
    NPCSpawnManager.clearWaveTracking(waveId);
    
    // 4. Next wave or complete raid
    if (isLastWave) {
        Bukkit.getScheduler().scheduleSyncDelayedTask(Main.instance, () -> {
            completeRaidWithEffects(session);
        }, 60L);
    } else {
        startWaveCountdown(session, waveId);
    }
}

### Wave Countdown System

**Location:** `raids/events/NPCDeathListener.java:160`

```java
private void startWaveCountdown(RaidSession session, String previousWaveId) {
    String countdownKey = session.getSessionId() + "_countdown";
    
    // Prevent duplicate countdowns
    if (activeCountdowns.contains(countdownKey)) return;
    activeCountdowns.add(countdownKey);
    
    List<Player> players = getActivePlayers(session);
    int nextWaveNumber = session.getCurrentWaveIndex() + 2;
    
    // Initial notification
    for (Player player : players) {
        player.sendMessage("");
        player.sendMessage("§6⏳ §fPreparándose para oleada §6" + nextWaveNumber + "§f...");
        player.sendMessage("");
    }
    
    final int[] countdown = {10}; // 10 second countdown
    final boolean[] hasStartedWave = {false};
    
    int taskId = Bukkit.getScheduler().scheduleSyncRepeatingTask(Main.instance, () -> {
        if (hasStartedWave[0]) return;
        
        List<Player> currentPlayers = getActivePlayers(session);
        
        if (countdown[0] > 0) {
            // Show countdown
            String msg = getCountdownMessage(countdown[0]);
            for (Player player : currentPlayers) {
                player.sendMessage(msg);
                if (countdown[0] <= 5) {
                    RaidSoundManager.playAlertSound(player);
                }
            }
            countdown[0]--;
        } else {
            // Start next wave
            hasStartedWave[0] = true;
            
            for (Player player : currentPlayers) {
                player.sendMessage("");
                player.sendMessage("§c§l⚔ ¡OLEADA " + nextWaveNumber + " INICIADA! ⚔");
                player.sendMessage("");
                RaidSoundManager.playWaveStartSound(player);
            }
            
            // Advance session
            session.moveToNextWave();
            Wave nextWave = session.getCurrentWave();
            
            if (nextWave != null) {
                nextWave.setStatus(WaveStatus.ACTIVE);
                String newWaveId = getWaveId(session);
                NPCSpawnManager.spawnWaveNpcs(nextWave, newWaveId);
            }
            
            // Cleanup
            activeCountdowns.remove(countdownKey);
            rewardsGiven.remove(previousWaveId);
            Bukkit.getScheduler().cancelTask(taskId);
        }
    }, 20L, 20L); // Run every second
    
    countdownTasks.put(countdownKey, taskId);
}
```

public class PartyManager {
    private static Map<UUID, Party> parties = new ConcurrentHashMap<>();
    private static Map<UUID, UUID> playerToParty = new ConcurrentHashMap<>();
    
    /**
     * Create a new party
     */
    public static Party createParty(Player leader) {
        if (playerToParty.containsKey(leader.getUniqueId())) {
            return null; // Already in a party
        }
        
        Party party = new Party();
        party.setPartyId(UUID.randomUUID());
        party.setLeader(leader.getUniqueId());
        party.getMembers().put(leader.getUniqueId(), leader.getName());
        party.getReadyStatus().put(leader.getUniqueId(), false);
        party.setStatus(PartyStatus.FORMING);
        party.setCreatedAt(System.currentTimeMillis());
        
        parties.put(party.getPartyId(), party);
        playerToParty.put(leader.getUniqueId(), party.getPartyId());
        
        return party;
    }
    
    /**
     * Invite player to party
     */
    public static boolean invitePlayer(Party party, Player inviter, Player target) {
        if (!party.isLeader(inviter.getUniqueId())) return false;
        if (playerToParty.containsKey(target.getUniqueId())) return false;
        
        party.getMembers().put(target.getUniqueId(), target.getName());
        party.getReadyStatus().put(target.getUniqueId(), false);
        playerToParty.put(target.getUniqueId(), party.getPartyId());
        
        return true;
    }
    
    /**
     * Set player ready status
     */
    public static void setReady(Party party, UUID playerId, boolean ready) {
        party.getReadyStatus().put(playerId, ready);
        
        if (party.allReady()) {
            party.setStatus(PartyStatus.READY);
        }
    }
}

private void executeWaveRewardsOnce(RaidSession session) {
    Wave wave = session.getCurrentWave();
    if (wave == null || !wave.hasRewards()) return;
    
    for (WaveReward reward : wave.getRewards()) {
        boolean shouldGive = reward.shouldExecute();
        
        if (shouldGive) {
            for (UUID playerId : session.getActivePlayers()) {
                Player player = Bukkit.getPlayer(playerId);
                if (player != null) {
                    // Replace placeholders
                    String command = reward.getCommand()
                        .replace("@p", player.getName())
                        .replace("@s", player.getName())
                        .replace("{player}", player.getName());
                    
                    // Execute as console
                    Bukkit.dispatchCommand(Bukkit.getConsoleSender(), command);

private static class LocationAdapter 
    implements JsonSerializer<Location>, JsonDeserializer<Location> {
    
    @Override
    public JsonElement serialize(Location loc, Type type, 
                                JsonSerializationContext context) {
        JsonObject obj = new JsonObject();
        obj.addProperty("world", loc.getWorld().getName());
        obj.addProperty("x", loc.getX());
        obj.addProperty("y", loc.getY());
        obj.addProperty("z", loc.getZ());
        obj.addProperty("yaw", loc.getYaw());
        obj.addProperty("pitch", loc.getPitch());
        return obj;
    }
    
    @Override
    public Location deserialize(JsonElement json, Type type,
                                JsonDeserializationContext context) {
        JsonObject obj = json.getAsJsonObject();
        String worldName = obj.get("world").getAsString();
        
        World world = Bukkit.getWorld(worldName);
        if (world == null) {
            System.err.println("[Raids] Mundo no encontrado: " + worldName);
            return null;
        }
        
        return new Location(
            world,
            obj.get("x").getAsDouble(),
            obj.get("y").getAsDouble(),
            obj.get("z").getAsDouble(),
            obj.get("yaw").getAsFloat(),
            obj.get("pitch").getAsFloat()
        );
    }
}

### CooldownManager

```java
public class CooldownManager {
    // Map: Raid ID → (Player UUID → Cooldown end time)
    private static Map<String, Map<UUID, Long>> cooldowns = new ConcurrentHashMap<>();
    
    /**
     * Set cooldown for player on specific raid
     */
    public static void setCooldown(String raidId, UUID playerId, long durationSeconds) {
        cooldowns.computeIfAbsent(raidId, k -> new ConcurrentHashMap<>())
            .put(playerId, System.currentTimeMillis() + (durationSeconds * 1000));
    }
    
    /**
     * Check if player has active cooldown
     */
    public static boolean hasCooldown(String raidId, UUID playerId) {
        Map<UUID, Long> raidCooldowns = cooldowns.get(raidId);
        if (raidCooldowns == null) return false;
        
        Long endTime = raidCooldowns.get(playerId);
        if (endTime == null) return false;
        
        if (System.currentTimeMillis() >= endTime) {
            raidCooldowns.remove(playerId);
            return false;
        }
        
        return true;
    }
    
    /**
     * Get remaining cooldown in seconds
     */
    public static long getRemainingCooldown(String raidId, UUID playerId) {
        Map<UUID, Long> raidCooldowns = cooldowns.get(raidId);
        if (raidCooldowns == null) return 0;
        
        Long endTime = raidCooldowns.get(playerId);
        if (endTime == null) return 0;
        
        long remaining = (endTime - System.currentTimeMillis()) / 1000;
        return Math.max(0, remaining);
    }
}
```

## Troubleshooting

<Accordion title="NPCs not spawning">
  **Solutions:**
  1. Verify CustomNPC is installed and loaded
  2. Check NPC name matches exactly (case-sensitive)
  3. Ensure spawn location world exists and is loaded
  4. Check console for CustomNPC API errors
  5. Verify arena spawn point is set: `/raid menu`
</Accordion>

<Accordion title="Wave not completing">
  **Solutions:**
  1. Check if all NPCs were properly tracked (console logs)
  2. Look for NPCs that fell out of world or despawned
  3. Admin can force complete: `/raid stop`
  4. Check `NPCSpawnManager` tracking maps for stuck entries
</Accordion>

<Accordion title="Rewards not executing">
  **Solutions:**
  1. Verify command syntax is valid
  2. Check chance value (0.0-1.0)
  3. Ensure placeholders are used: `@p` not literal player name
  4. Test command manually from console
  5. Check for command execution errors in console
</Accordion>
