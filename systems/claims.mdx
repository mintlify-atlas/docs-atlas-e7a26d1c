## Overview

The Claims System (also called Territory System) provides comprehensive land protection and management with **chunk-based territories**, **member permission roles**, **environmental customization**, and an advanced **leasing system** for sub-territory rentals. It integrates with Vault for economy transactions and supports complex ownership hierarchies.

```java
public class Terrain {
    
    public enum MemberRole {
        BUILD,      // Can place blocks
        BREAK,      // Can break blocks
        CONTAINERS, // Can open chests/furnaces
        ALL         // All permissions
    }
    
    private String id;                              // Unique identifier
    private int chunks;                             // Size in chunks
    private double price;                           // Purchase/sale price
    private UUID owner;                             // Owner UUID
    private String ownerName;                       // Owner name (cached)
    private Location origin;                        // Corner location
    private boolean committed;                      // Generated in world
    private Map<UUID, List<MemberRole>> members;    // Member permissions
    
    public int getSizeInBlocks() {
        return chunks * 16;
    }
    
    public boolean isInsideTerrain(Location loc) {
        if (origin == null) return false;
        if (!loc.getWorld().getName().equals(origin.getWorld().getName())) return false;
        
        int ox = origin.getBlockX();
        int oz = origin.getBlockZ();
        int size = getSizeInBlocks();
        int x = loc.getBlockX();
        int z = loc.getBlockZ();
        
        return x >= ox && x < ox + size && z >= oz && z < oz + size;
    }
    
    public boolean hasMemberRole(UUID uuid, MemberRole role) {
        if (!members.containsKey(uuid)) return false;
        List<MemberRole> roles = members.get(uuid);
        return roles.contains(MemberRole.ALL) || roles.contains(role);
    }
}
```

```java
public class TerrainManager {
    
    private static final int BUFFER_BLOCKS = 16;  // Minimum spacing between territories
    
    private final Map<String, Terrain> terrains = new HashMap<>();
    private final TerrainStorage storage;
    private Economy economy;  // Vault economy provider
    
    /**
     * Create a new territory (uncommitted)
     */
    public boolean createTerrain(String id, int chunks) {
        if (terrains.containsKey(id)) return false;
        
        Terrain terrain = new Terrain(id, chunks);
        terrains.put(id, terrain);
        storage.saveTerrain(terrain);
        
        return true;
    }
    
    /**
     * Commit territory at player location
     */
    public boolean commitTerrain(String id, Player player) {
        Terrain terrain = terrains.get(id);
        if (terrain == null || terrain.isCommitted()) return false;
        
        Location center = player.getLocation();
        int chunkX = center.getChunk().getX() * 16;
        int chunkZ = center.getChunk().getZ() * 16;
        int y = center.getBlockY();
        World world = center.getWorld();
        
        // Check for collisions with other territories
        String collision = checkCollision(id, world, chunkX, chunkZ, 
                                         terrain.getSizeInBlocks());
        if (collision != null) return false;
        
        // Set origin and commit
        Location origin = new Location(world, chunkX, y, chunkZ);
        terrain.setOrigin(origin);
        terrain.setCommitted(true);
        
        // Build visual borders and sign
        buildBorders(terrain, world, chunkX, chunkZ, y);
        placeSign(terrain, world, chunkX, chunkZ, y);
        
        storage.saveTerrain(terrain);
        return true;
    }
    
    /**
     * Check for collision with existing territories
     */
    public String checkCollision(String excludeId, World world, 
                                 int ox, int oz, int size) {
        int ax1 = ox - BUFFER_BLOCKS;
        int az1 = oz - BUFFER_BLOCKS;
        int ax2 = ox + size + BUFFER_BLOCKS;
        int az2 = oz + size + BUFFER_BLOCKS;
        
        for (Terrain other : terrains.values()) {
            if (!other.isCommitted()) continue;
            if (excludeId != null && other.getId().equals(excludeId)) continue;
            if (other.getOrigin() == null) continue;
            if (!other.getOrigin().getWorld().getName().equals(world.getName())) 
                continue;
            
            int bx1 = other.getOrigin().getBlockX();
            int bz1 = other.getOrigin().getBlockZ();
            int bx2 = bx1 + other.getSizeInBlocks();
            int bz2 = bz1 + other.getSizeInBlocks();
            
            boolean overlapX = ax1 < bx2 && ax2 > bx1;
            boolean overlapZ = az1 < bz2 && az2 > bz1;
            
            if (overlapX && overlapZ) return other.getId();
        }
        return null;
    }
}
```

```java
public class LeaseManager {
    
    public static final int MIN_CHUNKS_TO_LEASE = 4;       // Minimum parent size
    public static final int MIN_CYCLE_DAYS = 3;            // Minimum payment cycle
    public static final int MAX_CYCLE_DAYS = 20;           // Maximum payment cycle
    public static final int GRACE_PERIOD_DAYS = 3;         // Days before eviction
    public static final int PARENT_BORDER_MARGIN = 2;      // Blocks from parent edge
    
    /**
     * Owner offers lease to tenant
     */
    public LeaseContract offerContract(Terrain parentTerrain, Player owner,
                                      UUID tenantId, String tenantName,
                                      int chunks, double price, int cycleDays) {
        // Validate
        if (parentTerrain.getChunks() < MIN_CHUNKS_TO_LEASE) return null;
        if (cycleDays < MIN_CYCLE_DAYS || cycleDays > MAX_CYCLE_DAYS) return null;
        if (chunks < 1 || chunks >= parentTerrain.getChunks()) return null;
        
        // Generate unique contract ID
        String contractId = generateContractId(tenantName, parentTerrain.getId());
        
        // Create contract
        LeaseContract contract = new LeaseContract(
            contractId,
            parentTerrain.getId(),
            owner.getUniqueId(),
            owner.getName(),
            tenantId,
            tenantName,
            chunks,
            price,
            cycleDays,
            LeaseContract.ContractOrigin.OWNER_OFFER
        );
        
        contract.setStatus(LeaseContract.ContractStatus.PENDING_TENANT);
        contracts.put(contractId, contract);
        storage.saveContract(contract);
        
        return contract;
    }
    
    /**
     * Accept pending contract
     */
    public boolean acceptContract(LeaseContract contract, Player acceptor) {
        if (contract == null) return false;
        
        if (contract.getStatus() == LeaseContract.ContractStatus.PENDING_TENANT) {
            if (!contract.getTenantId().equals(acceptor.getUniqueId())) return false;
        } else if (contract.getStatus() == LeaseContract.ContractStatus.PENDING_OWNER) {
            if (!contract.getOwnerId().equals(acceptor.getUniqueId())) return false;
        } else {
            return false;
        }
        
        contract.setStatus(LeaseContract.ContractStatus.AWAITING_SUBTERRAIN);
        storage.saveContract(contract);
        return true;
    }
}
```

```java
public boolean assignSubTerrainAtPos(LeaseContract contract, Player owner,
                                    World world, int ox, int oz, int y) {
    if (contract.getStatus() != LeaseContract.ContractStatus.AWAITING_SUBTERRAIN) 
        return false;
    if (!contract.getOwnerId().equals(owner.getUniqueId())) return false;
    
    // Generate sub-territory ID
    String subId = generateSubTerrainId(contract.getParentTerrainId(), 
                                       contract.getTenantName());
    
    // Create sub-territory
    Terrain subTerrain = new Terrain(subId, contract.getChunks());
    subTerrain.setPrice(contract.getPricePerCycle());
    subTerrain.setOrigin(new Location(world, ox, y, oz));
    subTerrain.setCommitted(true);
    subTerrain.setOwner(contract.getTenantId());
    subTerrain.setOwnerName(contract.getTenantName());
    
    // Register and save
    TerrainManager.getInstance().getAll().put(subId, subTerrain);
    TerrainManager.getInstance().save(subTerrain);
    
    // Build red wool borders (distinct from parent)
    int size = contract.getChunks() * 16;
    buildSubBorders(world, ox, oz, y, size);
    placeSubSign(subTerrain, world, ox, oz, y, contract);
    
    // Link contract to sub-territory
    contract.setSubTerrainId(subId);
    
    // Process first payment
    Economy economy = TerrainManager.getInstance().getEconomy();
    if (economy != null && economy.has(contract.getTenantName(), 
                                       contract.getPricePerCycle())) {
        economy.withdrawPlayer(contract.getTenantName(), contract.getPricePerCycle());
        economy.depositPlayer(contract.getOwnerName(), contract.getPricePerCycle());
    }
    
    // Activate contract
    contract.activate();
    storage.saveContract(contract);
    
    return true;
}
```

### Automated Payment Processing

```java
private void processCycles() {
    List<LeaseContract> toProcess = new ArrayList<>(contracts.values());
    
    for (LeaseContract contract : toProcess) {
        if (contract.getStatus() == LeaseContract.ContractStatus.ACTIVE 
            && contract.isPaymentDue()) {
            processPayment(contract);
        } else if (contract.getStatus() == LeaseContract.ContractStatus.GRACE_PERIOD 
                   && contract.isGraceExpired()) {
            evictTenant(contract);
        }
    }
}

private void processPayment(LeaseContract contract) {
    Economy economy = TerrainManager.getInstance().getEconomy();
    if (economy == null) return;
    
    if (economy.has(contract.getTenantName(), contract.getPricePerCycle())) {
        // Payment successful
        economy.withdrawPlayer(contract.getTenantName(), contract.getPricePerCycle());
        economy.depositPlayer(contract.getOwnerName(), contract.getPricePerCycle());
        contract.renewPayment();
        storage.saveContract(contract);
        
        // Notify both parties
        notify(contract.getTenantName(), 
               "&7[Contrato] Pago de &f$" + (int) contract.getPricePerCycle() + 
               " &7debitado — &f" + contract.getSubTerrainId() + "&7.");
        notify(contract.getOwnerName(), 
               "&7[Contrato] Recibiste &f$" + (int) contract.getPricePerCycle() + 
               " &7de &f" + contract.getTenantName() + "&7.");
    } else {
        // Payment failed - start grace period
        contract.startGracePeriod();
        storage.saveContract(contract);
        
        notify(contract.getTenantName(), 
               "&c[Contrato] Sin fondos. Tienes &f" + GRACE_PERIOD_DAYS + 
               " días &cpara pagar o desalojar.");
        notify(contract.getOwnerName(), 
               "&c[Contrato] &f" + contract.getTenantName() + 
               " &cno pudo pagar. Período de gracia iniciado.");
    }
}

private void evictTenant(LeaseContract contract) {
    // Remove sub-territory
    if (contract.getSubTerrainId() != null) {
        removeSubBorders(contract.getSubTerrainId());
        TerrainManager.getInstance().deleteTerrain(contract.getSubTerrainId());
    }
    
    contract.setStatus(LeaseContract.ContractStatus.EXPIRED);
    storage.saveContract(contract);
    
    // Notify both parties
    notify(contract.getTenantName(), 
           "&c[Contrato] Período de gracia expirado. Desalojado de &f" + 
           contract.getSubTerrainId() + "&c.");
    notify(contract.getOwnerName(), 
           "&7[Contrato] &f" + contract.getTenantName() + 
           " &7fue desalojado por falta de pago.");
}
```

```java
public boolean acceptMove(String contractId, Player tenant) {
    LeaseContract contract = contracts.get(contractId);
    if (contract == null || !contract.getTenantId().equals(tenant.getUniqueId())) 
        return false;
    if (!contract.isHasPendingMove()) return false;
    
    World world = Bukkit.getWorld(contract.getPendingMoveWorld());
    if (world == null) {
        contract.clearPendingMove();
        storage.saveContract(contract);
        return false;
    }
    
    // Validate new position
    int ox = contract.getPendingMoveX1();
    int oz = contract.getPendingMoveZ1();
    int size = contract.getChunks() * 16;
    
    String valid = validateSelectionForMove(contract, world, ox, oz, size);
    if (valid != null) {
        contract.clearPendingMove();
        storage.saveContract(contract);
        tenant.sendMessage(CC.translate("&cEl traslado ya no es válido: &7" + valid));
        return false;
    }
    
    // Execute move
    Terrain sub = TerrainManager.getInstance().getTerrain(contract.getSubTerrainId());
    int y = sub.getOrigin() != null ? sub.getOrigin().getBlockY() 
                                    : tenant.getLocation().getBlockY();
    
    doMoveSubTerrain(contract, world, ox, oz, y);
    contract.clearPendingMove();
    storage.saveContract(contract);
    
    return true;
}
```

```java
public class TerrainCustomizeManager {
    
    /**
     * Change territory biome
     */
    public static boolean setBiome(final Terrain terrain, final Player player, 
                                  final String biomeName) {
        final Biome biome;
        try {
            biome = Biome.valueOf(biomeName.toUpperCase());
        } catch (IllegalArgumentException e) {
            player.sendMessage(CC.translate("&cBioma desconocido: &f" + biomeName));
            return false;
        }
        
        final World world = terrain.getOrigin().getWorld();
        final int ox = terrain.getOrigin().getBlockX();
        final int oz = terrain.getOrigin().getBlockZ();
        final int size = terrain.getSizeInBlocks();
        
        // Run asynchronously for performance
        Bukkit.getScheduler().runTaskAsynchronously(Main.instance, () -> {
            for (int x = ox; x < ox + size; x++) {
                for (int z = oz; z < oz + size; z++) {
                    world.setBiome(x, z, biome);
                }
            }
            
            // Refresh chunks on main thread
            Bukkit.getScheduler().runTask(Main.instance, () -> {
                for (int cx = ox >> 4; cx <= (ox + size) >> 4; cx++) {
                    for (int cz = oz >> 4; cz <= (oz + size) >> 4; cz++) {
                        world.refreshChunk(cx, cz);
                    }
                }
                player.sendMessage(CC.translate(
                    "&7Bioma del terreno &f" + terrain.getId() + 
                    " &7cambiado a &f" + biomeName + "&7."));
            });
        });
        
        return true;
    }
    
    /**
     * Set territory time (player-specific)
     */
    public static void setTime(Terrain terrain, Player player, long ticks) {
        terrainTime.put(terrain.getId(), ticks);
        storage.setTime(terrain.getId(), ticks);
        player.setPlayerTime(ticks, false);
        player.sendMessage(CC.translate("&7Tiempo del terreno configurado."));
    }
    
    /**
     * Set territory weather
     */
    public static void setWeather(Terrain terrain, Player player, String type) {
        World world = terrain.getOrigin().getWorld();
        terrainWeather.put(terrain.getId(), type);
        storage.setWeather(terrain.getId(), type);
        
        switch (type) {
            case "CLEAR":
                world.setStorm(false);
                world.setThundering(false);
                world.setWeatherDuration(Integer.MAX_VALUE);
                break;
            case "RAIN":
                world.setStorm(true);
                world.setThundering(false);
                world.setWeatherDuration(Integer.MAX_VALUE);
                break;
            case "STORM":
                world.setStorm(true);
                world.setThundering(true);
                world.setThunderDuration(Integer.MAX_VALUE);
                world.setWeatherDuration(Integer.MAX_VALUE);
                break;
        }
    }
}
```

```java
public static void applyEffectToPlayer(Player player) {
    Terrain terrain = TerrainManager.getInstance().getTerrainAt(player.getLocation());
    if (terrain == null) return;
    
    String effectName = getEffect(terrain);
    if ("NONE".equals(effectName) || effectName.isEmpty()) return;
    
    PotionEffectType pet = getPotionEffectType(effectName);
    if (pet == null) return;
    
    // Check if player already has effect
    PotionEffect existing = null;
    for (PotionEffect pe : player.getActivePotionEffects()) {
        if (pe.getType().equals(pet)) {
            existing = pe;
            break;
        }
    }
    
    // Don't reapply if duration > 100 ticks
    if (existing != null && existing.getDuration() > 100) return;
    
    // Apply effect (5 seconds, level 1, ambient)
    player.addPotionEffect(new PotionEffect(pet, 300, 0, true), true);
}

private static PotionEffectType getPotionEffectType(String name) {
    switch (name) {
        case "REGENERATION": return PotionEffectType.REGENERATION;
        case "SPEED": return PotionEffectType.SPEED;
        case "INCREASE_DAMAGE": return PotionEffectType.INCREASE_DAMAGE;
        case "WATER_BREATHING": return PotionEffectType.WATER_BREATHING;
        case "NIGHT_VISION": return PotionEffectType.NIGHT_VISION;
        case "INVISIBILITY": return PotionEffectType.INVISIBILITY;
        case "SLOW": return PotionEffectType.SLOW;
        default: return null;
    }
}
```

  - `/terrain delete <id>` - Delete territory
  - `/terrain setprice <id> <price>` - Set purchase price
  - `/terrain list` - List all territories
  - `/terrain info <id>` - View territory details
  
  **Owner Commands:**
  - `/terrain buy` - Purchase territory you're standing in
  - `/terrain sell` - Dissolve ownership (doesn't delete territory)
  - `/terrain addmember <player> <role>` - Add member with role
  - `/terrain removemember <player>` - Remove member
  - `/terrain customize` - Open customization GUI
</Accordion>

## Troubleshooting

<Accordion title="Territory borders not appearing">
  **Solutions:**
  1. Verify territory is committed: `/terrain info <id>`
  2. Check if blocks were replaced by players
  3. Recommit territory to regenerate borders
  4. Ensure world is loaded when committing
</Accordion>

<Accordion title="Lease payments not processing">
  **Solutions:**
  1. Verify Vault is installed and economy provider is active
  2. Check tenant has sufficient funds
  3. Review payment scheduler in console logs
  4. Manually trigger payment: `/lease pay`
  5. Check contract status: `/lease list`
</Accordion>

<Accordion title="Biome changes not visible">
  **Solutions:**
  1. Players must relog or move out and back into territory
  2. Force chunk reload: `/terrain customize` → Biome → reselect biome
  3. Check console for async task completion
  4. Verify biome name is valid Minecraft biome
</Accordion>
